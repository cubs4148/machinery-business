const carousel1Images = document.querySelectorAll('.carousel:nth-of-type(1) .carousel-images img'); const carousel2Images = document.querySelectorAll('.carousel:nth-of-type(2) .carousel-images img'); const prevButton1 = document.getElementById('prev'); const nextButton1 = document.getElementById('next'); const prevButton2 = document.getElementById('prev2'); const nextButton2 = document.getElementById('next2'); let currentIndex1 = 0; let currentIndex2 = 0; function updateCarousel(images, currentIndex) { images.forEach((img, index) => { img.classList.toggle('active', index === currentIndex); }); } // Event Listeners for Carousel 1 nextButton1.addEventListener('click', () => { currentIndex1 = (currentIndex1 + 1) % carousel1Images.length; updateCarousel(carousel1Images, currentIndex1); }); prevButton1.addEventListener('click', () => { currentIndex1 = (currentIndex1 - 1 + carousel1Images.length) % carousel1Images.length; updateCarousel(carousel1Images, currentIndex1); }); // Event Listeners for Carousel 2 nextButton2.addEventListener('click', () => { currentIndex2 = (currentIndex2 + 1) % carousel2Images.length; updateCarousel(carousel2Images, currentIndex2); }); prevButton2.addEventListener('click', () => { currentIndex2 = (currentIndex2 - 1 + carousel2Images.length) % carousel2Images.length; updateCarousel(carousel2Images, currentIndex2); }); // Initialize Active Images updateCarousel(carousel1Images, currentIndex1); updateCarousel(carousel2Images, currentIndex2);
